<project_specification>
  <project_name>Lead Awaker Mobile Update</project_name>

  <overview>
    A comprehensive mobile UI overhaul of the Lead Awaker CRM platform — an AI-powered WhatsApp lead reactivation engine built with React, Express, and PostgreSQL. This update transforms the existing desktop-first application into a polished, native-feeling mobile experience with thumb-friendly navigation, full-screen detail panels with tabs, swipe gestures, haptic feedback, and faux-glassmorphism visual polish. The existing backend, database, and API remain untouched — this is a purely frontend/UI project targeting screens under 768px.
  </overview>

  <technology_stack>
    <frontend>
      <framework>React (Vite) — existing codebase</framework>
      <styling>Tailwind CSS v4 with CSS custom properties design system</styling>
      <animations>Framer Motion (existing), CSS @keyframes, Web Animations API for haptics</animations>
      <state>React Query (TanStack), React hooks, localStorage persistence</state>
      <routing>Wouter</routing>
      <i18n>react-i18next (en/pt/nl)</i18n>
      <ui_library>shadcn/ui, Radix UI primitives</ui_library>
    </frontend>
    <backend>
      <runtime>Node.js + Express (unchanged)</runtime>
      <database>PostgreSQL with Drizzle ORM (unchanged)</database>
    </backend>
    <communication>
      <api>REST API (unchanged)</api>
    </communication>
  </technology_stack>

  <ui_layout>
    <main_structure>
      Mobile layout (below 768px):
      - Top bar: fixed, height 56px + safe-area-inset-top. Contains: swipable KPI strip (left), bell icon, support chat icon, user avatar (right)
      - Main content: full-screen, scrollable, padding-top for topbar, padding-bottom for bottom bar
      - Bottom bar: fixed, height 64px + safe-area-inset-bottom. Contains: 5 navigation icons with labels

      Desktop layout (768px and above):
      - Unchanged. Left sidebar + topbar + main content area remains identical.
    </main_structure>

    <mobile_navigation_pattern>
      List pages use full-screen card lists. Tapping a card navigates to a full-screen detail view with tabs (not scroll sections). Back button returns to list. This pattern applies to: Leads, Campaigns, Conversations, Billing, Tasks, Accounts.
    </mobile_navigation_pattern>

    <mobile_detail_panel_pattern>
      Detail panels are full-screen takeovers with:
      - Fixed header (entity name + back button)
      - Tab bar below header (e.g., Info | Chat | Activity | Notes)
      - Scrollable tab content
      - Faux-glassmorphism blur on header when content scrolls beneath
      - No backdrop dialogs (panel-first rule still applies)
    </mobile_detail_panel_pattern>

    <mobile_filter_pattern>
      All list pages use a single filter icon that opens a bottom sheet containing:
      - Search input
      - Group by options
      - Sort by options
      - Filter options (status, account, campaign)
      - Clear all / Apply buttons
    </mobile_filter_pattern>
  </ui_layout>

  <design_system>
    <color_palette>
      All existing CSS custom properties and brand tokens apply unchanged:
      - --brand-indigo: #4F46E5 (primary actions, bottom bar active state)
      - --brand-yellow: #FFDF3D (KPI accent, agency avatar ring)
      - --highlight-active: #FFF28D (active tab pill)

      - Dark mode tokens unchanged
    </color_palette>

    <typography>
      - Outfit for headings (font-heading)
      - Inter for body text (font-sans)
      - Mobile font sizes: slightly larger touch-friendly labels on bottom bar (10px), standard body (14px), headings (18-24px)
    </typography>

    <mobile_specific_design>
      - Faux-glassmorphism: backdrop-filter: blur(12px) saturate(1.2) on sticky headers/footers when content scrolls beneath
      - Touch targets: minimum 44x44px on all interactive elements
      - Bottom bar icons: larger (20-24px) with 10px labels beneath
      - Top bar icons: standard 16px (h-4 w-4) in 36px touch circles
      - Card gaps: 3px vertical gap (matching desktop pattern)
      - Border radius: 1.5rem (24px) global radius (matching desktop)
      - Swipe gesture threshold: 60px horizontal to trigger action
      - Pull-to-refresh: 60px pull distance to trigger refresh
    </mobile_specific_design>
  </design_system>
