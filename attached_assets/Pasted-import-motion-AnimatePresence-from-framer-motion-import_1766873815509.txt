import { motion, AnimatePresence } from "framer-motion";
import { useState, useEffect } from "react";

export default function ChatCard2D() {
  const [isPlaying, setIsPlaying] = useState(false);
  const [currentStep, setCurrentStep] = useState(0);

  const totalSteps = 12; // Total messages + typing indicators
  const totalDuration = 15; // 15 seconds total

  useEffect(() => {
    if (isPlaying && currentStep < totalSteps) {
      const stepDuration = totalDuration / totalSteps;
      const timeout = setTimeout(() => {
        setCurrentStep((prev) => prev + 1);
      }, stepDuration * 1000);

      return () => clearTimeout(timeout);
    }
  }, [isPlaying, currentStep]);

  const messages = [
    // Initial static messages (0-5)
    { type: "sophie", text: "Hi, this is Sophie from Sterling Finance, is this Jack who wanted to check if they were owed a refund on their car finance?", time: "14:35" },
    { type: "sophie", text: "just bumping this up in case you got busy before :)", time: "14:55" },
    { type: "jack", text: "Hi, yes", time: "15:00" },
    { type: "jack", text: "I would like to check. I had 2 cars on finance 5 years ago", time: "15:01" },
    { type: "sophie", text: "That's great Jack! Do you happen to have your vehicle finance agreement documents handy for those cars?", time: "15:04" },
    
    // New animated messages (6-17)
    { type: "jack", text: "No sorry,", time: "15:05", isSplit: true },
    { type: "jack", text: "Not at Hand", time: "15:05", isSplit: true },
    { type: "sophie", text: "No worries at all Jack. We can actually submit a Data Subject Access Request (DSAR) on your behalf to get those documents for you. Could you please complete this short DSAR form by clicking on this URL https://www.dsarform.com/lead315? It will help us move things forward with you", time: "15:09" },
    { type: "jack", text: "What is the process from here?", time: "15:20", isSplit: true },
    { type: "jack", text: "Do you let me know if I'm elegible to make a claim once you retrieve the docs and then I confirm I want to proceed?", time: "15:20", isSplit: true },
    { type: "sophie", text: "Absolutely, Jack. Once you've completed the DSAR form, we'll request your vehicle finance agreement and review it. If we find that you're elegible for a claim, we'll get in touch with you to discuss the next steps. It typically takes us about 30 workind days to get everything sorted. Remember, we operate on a no-win no fee basis, so there's no risk to you. Just pop over to the URL and fill out the form when you can https://www.dsarform.com/lead315", time: "15:22" },
    { type: "jack", text: "Thanks.", time: "15:24" }
  ];

  const messageVariants = {
    hidden: { opacity: 0, y: 10, scale: 0.95 },
    visible: { 
      opacity: 1, 
      y: 0, 
      scale: 1,
      transition: { duration: 0.4, ease: "easeOut" }
    }
  };

  const containerVariants = {
    hidden: { opacity: 0 },
    visible: {
      opacity: 1,
      transition: {
        staggerChildren: 0.1,
        delayChildren: 0.2
      }
    }
  };

  return (
    <div className="bg-white rounded-2xl shadow-xl border border-slate-100 overflow-hidden">
      {/* Chat Header */}
      <div className="p-4 flex items-center gap-3" style={{ backgroundColor: "#2563EB" }}>
        <div className="rounded-full flex items-center justify-center text-white font-bold text-sm" style={{ width: "44px", height: "44px", backgroundColor: "#6B6B6B" }}>
          J
        </div>
        <div>
          <h3 className="text-white font-medium text-sm">Jack Johnson</h3>
          <p className="text-xs flex items-center gap-1" style={{ color: "#FCC700" }}>
            <span className="w-1.5 h-1.5 rounded-full animate-pulse" style={{ backgroundColor: "#FCC700" }} />
            Online
          </p>
        </div>
      </div>

      {/* Chat Messages */}
      <div className="p-6 space-y-4 bg-slate-50 min-h-[500px] relative overflow-hidden">
        <div className="absolute inset-0 bg-gradient-to-br from-slate-100/20 via-transparent to-slate-100/20 pointer-events-none" />
        
        <div className="relative z-10 space-y-4 max-h-[400px] overflow-y-auto">
          <motion.div 
            className="space-y-4"
            variants={containerVariants}
            initial="hidden"
            animate={currentStep > 0 ? "visible" : "hidden"}
          >
            {/* Render all messages up to current step */}
            {messages.slice(0, Math.min(currentStep + 1, messages.length)).map((message, index) => (
              <motion.div
                key={`${message.type}-${index}`}
                className={`flex ${message.type === 'sophie' ? 'justify-end' : 'justify-start'}`}
                custom={index}
                variants={messageVariants}
                initial="hidden"
                animate={index <= currentStep ? "visible" : "hidden"}
                style={{ opacity: index <= currentStep ? 1 : 0 }}
              >
                <div className={`flex flex-col ${message.type === 'sophie' ? 'items-end' : 'items-start'} gap-1`}>
                  <div 
                    className={`px-4 py-3 max-w-[85%] shadow-sm text-sm rounded-2xl ${

                        ? 'text-white rounded-tr-sm bg-gradient-to-r from-[#2563EB] to-[#3B82F6]'

                      <a 
                        href="https://www.dsarform.com/lead315"
                        target="_blank"
                        rel="noopener noreferrer"
                        className="text-yellow-200 hover:text-yellow-100 underline font-medium block mt-1"
                      >
                        https://www.dsarform.com/lead315 →
                      </a>
                    )}
                  </div>
                  <span className={`text-xs text-slate-400 ${message.type === 'sophie' ? 'pr-2' : 'pl-2'}`}>
                    {message.time}
                  </span>
                </div>
              </motion.div>
            ))}

            {/* Typing indicator when not finished */}
            {isPlaying && currentStep < totalSteps && currentStep >= 5 && (
              <motion.div 
                className="flex justify-start"
                initial={{ opacity: 0, scale: 0.8 }}
                animate={{ 
                  opacity: 1, 
                  scale: 1 
                }}
                exit={{ opacity: 0, scale: 0.8 }}
                transition={{ duration: 0.3 }}
              >
                <div className="bg-slate-200 text-slate-500 rounded-full px-4 py-2 shadow-sm flex items-center justify-center gap-2">
                  <motion.span 
                    className="text-4xl font-black" 
                    animate={{ y: [0, -8, 0] }} 
                    transition={{ duration: 1.2, repeat: Infinity, delay: 0 }} 
                    style={{ color: '#6B7280' }}
                  >
                    ·
                  </motion.span>
                  <motion.span 
                    className="text-4xl font-black" 
                    animate={{ y: [0, -8, 0] }} 
                    transition={{ duration: 1.2, repeat: Infinity, delay: 0.2 }} 
                    style={{ color: '#6B7280' }}
                  >
                    ·
                  </motion.span>
                  <motion.span 
                    className="text-4xl font-black" 
                    animate={{ y: [0, -8, 0] }} 
                    transition={{ duration: 1.2, repeat: Infinity, delay: 0.4 }} 
                    style={{ color: '#6B7280' }}
                  >
                    ·
                  </motion.span>
                </div>
              </motion.div>
            )}
          </motion.div>
        </div>

        {/* Start/Reset Button */}
        {!isPlaying ? (
          <motion.div

            initial={{ opacity: 0, y: 20 }}
            animate={{ opacity: 1, y: 0 }}
            transition={{ delay: 1 }}
          >
            <button
              onClick={() => {
                setCurrentStep(5); // Start from after last static message
                setIsPlaying(true);
              }}
              className="w-full bg-gradient-to-r from-[#2563EB] to-[#3B82F6] text-white py-3 px-6 rounded-xl font-medium text-sm shadow-lg hover:shadow-xl transition-all duration-200 flex items-center justify-center gap-2"
            >
              <svg className="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M14.752 11.168l-3.197-2.132A1 1 0 0010 9.87v4.263a1 1 0 001.555.832l3.197-2.132a1 1 0 000-1.664z" />
              </svg>
              Watch the conversation continue
            </button>
          </motion.div>
        ) : currentStep >= totalSteps ? (
          <motion.div 
            className="pt-4 border-t border-slate-200"
            initial={{ scale: 0.9, opacity: 0 }}
            animate={{ scale: 1, opacity: 1 }}
            transition={{ delay: 0.5 }}
          >
            <button
              onClick={() => {
                setCurrentStep(0);
                setIsPlaying(false);
              }}
              className="w-full bg-emerald-500 hover:bg-emerald-600 text-white py-3 px-6 rounded-xl font-medium text-sm shadow-lg hover:shadow-xl transition-all duration-200 flex items-center justify-center gap-2"
            >
              <svg className="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M4 4v5h.582m15.356 2A8.001 8.001 0 004.582 9m0 0H9m11 11v-5h-.581m0 0a8.003 8.003 0 01-15.357-2m15.357 2H15" />
              </svg>
              Replay conversation
            </button>
          </motion.div>
        ) : null}
      </div>
    </div>
  );
}This enhanced version adds:Start Button - Triggers the animation sequence from where it left off15-second Timeline - Each message appears precisely timed across 15 seconds totalSmooth Animations - Messages slide in with scale and fade effectsTyping Indicators - Show between messages for realismSplit Messages - Jack's multi-line responses appear as separate messagesClickable URL - DSAR form link is properly formatted and clickableReplay Functionality - Reset and watch againResponsive Design - Scrollable chat area with improved gradientsThe animation feels natural and professional, perfect for your services page!
