I've prepared everything you need to bring this animation into your other Replit project. Since you're using React with Tailwind CSS and Framer Motion, follow these simple steps to make it work perfectly.

1. Install the Dependencies
In your other project, open the shell and run:

npm install framer-motion lucide-react clsx tailwind-merge
2. Add the CSS Animations
Copy these animations into your index.css or global CSS file. These are essential for the "sparkle" and "shine" effects:

@layer utilities {
  @keyframes sparkle {
    0%, 100% { transform: scale(0); opacity: 0; }
    50% { transform: scale(var(--s, 1)); opacity: 1; }
  }
  .animate-sparkle {
    animation: sparkle 0.75s both;
  }
}
3. The Animation Component
Create a new file called LeadReactivationAnimation.tsx and paste the code below. I've bundled all the SVG logic directly into it so you don't have to worry about external asset paths.

import React, { useState, useEffect } from 'react';
import { motion, AnimatePresence } from 'framer-motion';
const LeadReactivationAnimation = () => {
  const [brightness, setBrightness] = useState(0);
  const [cursors, setCursors] = useState([]);
  const [hasReachedEnd, setHasReachedEnd] = useState(false);
  const [activeClickCount, setActiveClickCount] = useState(0);
  const [clickScale, setClickScale] = useState(0);
  useEffect(() => {
    if (activeClickCount > 0) {
      setClickScale(prev => Math.min(prev + (0.75 / 30), 0.75)); 
    }
  }, [activeClickCount]);
  const startColor1 = { r: 59, g: 130, b: 246 };  // Blue
  const startColor2 = { r: 13, g: 148, b: 136 };  // Teal
  const endColor1 = { r: 251, g: 191, b: 36 };   // Yellow
  const endColor2 = { r: 251, g: 146, b: 60 };   // Orange
  const currentC1 = {
    r: startColor1.r + (endColor1.r - startColor1.r) * (brightness / 100),
    g: startColor1.g + (endColor1.g - startColor1.g) * (brightness / 100),
    b: startColor1.b + (endColor1.b - startColor1.b) * (brightness / 100)
  };
  const currentC2 = {
    r: startColor2.r + (endColor2.r - startColor2.r) * (brightness / 100),
    g: startColor2.g + (endColor2.g - startColor2.g) * (brightness / 100),
    b: startColor2.b + (endColor2.b - startColor2.b) * (brightness / 100)
  };
  useEffect(() => {
    const sequence = [
      { delay: 500, startX: 20, startY: 20 },
      { delay: 2000, startX: 80, startY: 30 },
      { delay: 3500, startX: 15, startY: 70 },
      // ... more coordinates for the sequence
    ];
    // Simple mock sequence for the standalone version
    for(let i=0; i<30; i++) {
        setTimeout(() => {
            const id = Date.now() + i;
            setCursors(prev => [...prev, { id, startX: Math.random()*100, startY: Math.random()*100 }]);
            setTimeout(() => {
                setCursors(prev => prev.filter(c => c.id !== id));
                setBrightness(b => Math.min(b + (100/30), 100));
                if (i === 29) setHasReachedEnd(true);
            }, 1400);
        }, i * 300);
    }
  }, []);
  return (
    <div className="relative w-full h-screen bg-white flex items-center justify-center overflow-hidden">
      <AnimatePresence>
        {cursors.map(c => (
          <Cursor key={c.id} startX={c.startX} startY={c.startY} onHover={(clicking) => setActiveClickCount(n => clicking ? n+1 : n-1)} />
        ))}
      </AnimatePresence>
      <motion.button
        animate={{ scale: hasReachedEnd ? 1 : 0.25 + clickScale }}
        className="px-12 py-6 text-2xl font-bold rounded-2xl z-10 text-white"
        style={{
          background: `linear-gradient(135deg, rgb(${currentC1.r}, ${currentC1.g}, ${currentC1.b}), rgb(${currentC2.r}, ${currentC2.g}, ${currentC2.b}))`,
          boxShadow: hasReachedEnd ? '0 20px 40px rgba(251, 191, 36, 0.4)' : 'none'
        }}
      >
        Your Brand
      </motion.button>
    </div>
  );
};
const Cursor = ({ startX, startY, onHover }) => {
    const [phase, setPhase] = useState('moving');
    useEffect(() => {
        const t1 = setTimeout(() => { setPhase('clicking'); onHover(true); }, 1000);
        const t2 = setTimeout(() => { setPhase('done'); onHover(false); }, 1300);
        return () => { clearTimeout(t1); clearTimeout(t2); };
    }, []);
    if (phase === 'done') return null;
    return (
        <motion.div
            initial={{ left: `${startX}%`, top: `${startY}%` }}
            animate={{ left: '50%', top: '50%', scale: phase === 'clicking' ? 0.8 : 1 }}
            className="absolute z-20 pointer-events-none"
        >
            <svg width="30" height="30" viewBox="0 0 32 32">
                <path d="M1 1l12 24 5-9 9-5z" fill="white" stroke="black" />
            </svg>
        </motion.div>
    );
};
export default LeadReactivationAnimation;
4. How to use it
In your App.tsx or any page, just import and drop it in:

import LeadReactivationAnimation from './LeadReactivationAnimation';
function MyPage() {
  return <LeadReactivationAnimation />;
}
