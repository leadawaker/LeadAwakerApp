<project_specification>
  <project_name>Lead Awaker Agents</project_name>

  <overview>
    An evolution of the existing AI agent chat system inside the LeadAwaker CRM. Brings a full Claude Code-like experience — streaming responses, model switching, thinking toggles, skills, file uploads, voice memos, and page-aware context — into the CRM as embedded AI assistants. Agents can read and modify CRM data, access Google Workspace via GOG, and persist conversations across sessions. The chat widget stays open across page navigation, giving the admin a persistent AI co-pilot while browsing the CRM.
  </overview>

  <technology_stack>
    <frontend>
      <framework>React (TypeScript) — existing LeadAwaker frontend</framework>
      <styling>Tailwind CSS — existing design system</styling>
      <state>Wouter routing, React context/hooks</state>
      <streaming>Server-Sent Events (SSE) for token-by-token streaming</streaming>
    </frontend>
    <backend>
      <runtime>Node.js / Express — existing LeadAwaker backend</runtime>
      <database>PostgreSQL — existing database, extended with agent tables</database>
      <ai_provider>Anthropic Claude API (user's own subscription key)</ai_provider>
      <integrations>GOG CLI for Google Docs/Sheets/Gmail access</integrations>
    </backend>
    <communication>
      <api>REST API + SSE streaming for chat responses</api>
    </communication>
  </technology_stack>

  <database_schema>
    <tables>
      <ai_agents>
        - id (uuid, primary key)
        - name (text)
        - icon (text, emoji or icon identifier)
        - system_prompt_id (uuid, FK to prompts library, nullable)
        - default_system_prompt (text, Code Runner base prompt)
        - model (text, default 'claude-sonnet-4-20250514')
        - thinking_level (text, default 'medium')
        - permissions (jsonb, e.g. {"read": true, "write": true, "delete": false})
        - page_awareness_enabled (boolean, default true)
        - created_by (uuid, FK to users)
        - created_at (timestamp)
        - updated_at (timestamp)
      </ai_agents>

      <agent_conversations>
        - id (uuid, primary key)
        - agent_id (uuid, FK to ai_agents)
        - user_id (uuid, FK to users)
        - title (text, AI-generated)
        - model (text, model used)
        - thinking_level (text)
        - total_input_tokens (integer, default 0)
        - total_output_tokens (integer, default 0)
        - is_active (boolean, default true)
        - created_at (timestamp)
        - updated_at (timestamp)
      </agent_conversations>

      <agent_messages>
        - id (uuid, primary key)
        - conversation_id (uuid, FK to agent_conversations)
        - role (text, 'user' or 'assistant')
        - content (text)
        - metadata (jsonb, for token counts, model info, thinking data)
        - attachments (jsonb, array of file references)
        - page_context (jsonb, route + data snapshot at time of message)
        - created_at (timestamp)
      </agent_messages>

      <agent_files>
        - id (uuid, primary key)
        - conversation_id (uuid, FK to agent_conversations)
        - message_id (uuid, FK to agent_messages)
        - filename (text)
        - mime_type (text)
        - file_path (text)
        - file_size (integer)
        - transcription (text, for voice memos)
        - created_at (timestamp)
      </agent_files>
    </tables>
  </database_schema>

  <api_endpoints_summary>
    <agents>
      - GET /api/agents (list all agents)
      - POST /api/agents (create new agent, clone from template)
      - GET /api/agents/:id (get agent details)
      - PUT /api/agents/:id (update agent settings)
      - DELETE /api/agents/:id (delete agent)
    </agents>
    <conversations>
      - GET /api/agents/:id/conversations (list conversations for agent)
      - POST /api/agents/:id/conversations (create new conversation)
      - GET /api/agent-conversations/:id (get conversation with messages)
      - DELETE /api/agent-conversations/:id (delete/clear conversation)
    </conversations>
    <messages>
      - POST /api/agent-conversations/:id/messages (send message, returns SSE stream)
      - POST /api/agent-conversations/:id/files (upload file attachment)
    </messages>
    <voice>
      - POST /api/agent-voice/transcribe (transcribe voice memo audio)
    </voice>
    <context>
      - POST /api/agent-context/page (submit current page context)
    </context>
    <skills>
      - GET /api/agent-skills (list available Claude Code skills)
      - POST /api/agent-skills/execute (execute a skill)
    </skills>
  </api_endpoints_summary>
